Smart Model Router

Intelligent routing system that automatically selects the optimal AI model based on task requirements, cost constraints, and performance metrics.

Selection Criteria

  1. Content Type Analysis: Detect whether input is text, image, audio, file, or multimodal combination
  2. Task Complexity Assessment: Evaluate if it's a simple query or requires deep reasoning
  3. Latency Requirements: Determine need for real-time vs batch processing
  4. Budget Optimization: Balance token costs against quality needs
  5. Domain Specialization: Route to models optimized for code, math, creative, or analytical tasks
  6. Fallback Strategy: Maintain backup models for reliability

# services/smart_router.py
from typing import Dict, Any, List
from enum import Enum

class TaskType(Enum):
    TEXT_SIMPLE = "text_simple"
    TEXT_COMPLEX = "text_complex"
    CODE_GENERATION = "code_generation"
    CODE_DEBUGGING = "code_debugging"
    IMAGE_ANALYSIS = "image_analysis"
    IMAGE_GENERATION = "image_generation"
    AUDIO_TRANSCRIPTION = "audio_transcription"
    FILE_PROCESSING = "file_processing"
    MULTIMODAL = "multimodal"

class SmartRouter:
    def __init__(self, config: Dict[str, Any]):
        self.models = config['models']
        self.rules = config['routing_rules']
        self.cost_tracker = CostTracker()
        self.performance_monitor = PerformanceMonitor()
    
    async def route(self, message: UniversalMessage) -> RoutingDecision:
        """Analyze message and select optimal model(s)"""
        
        # Step 1: Classify task type
        task_type = await self._classify_task(message)
        
        # Step 2: Analyze complexity
        complexity = self._analyze_complexity(message)
        
        # Step 3: Check constraints (budget, latency, etc.)
        constraints = await self._get_constraints(message.user_id)
        
        # Step 4: Select candidates based on rules
        candidates = self._get_candidates(task_type, complexity)
        
        # Step 5: Optimize for cost/performance
        selected = self._optimize_selection(
            candidates,
            constraints,
            self.performance_monitor.get_stats()
        )
        
        return RoutingDecision(
            primary_model=selected['primary'],
            fallback_models=selected['fallbacks'],
            pipeline=selected['pipeline'],
            estimated_cost=selected['cost'],
            estimated_latency=selected['latency']
        )

# services/memory_manager.py
from qdrant_client import QdrantClient
from datetime import datetime, timedelta
import numpy as np

class MemoryManager:
    def __init__(self, vector_db: QdrantClient, pg_db):
        self.vector_db = vector_db
        self.pg_db = pg_db
        self.embedder = EmbeddingModel('text-embedding-3-large')
    
    async def retrieve_context(
        self,
        query: str,
        user_id: str,
        max_tokens: int = 4000,
        time_decay_days: int = 30
    ) -> List[Memory]:
        """Retrieve relevant memories with time-based decay"""
        
        # Generate query embedding
        query_embedding = await self.embedder.embed(query)
        
        # Semantic search in vector database
        results = self.vector_db.search(
            collection='memories',
            query_vector=query_embedding,
            query_filter=Filter(must=[
                FieldCondition(key='user_id', match=MatchValue(value=user_id))
            ]),
            limit=20,
            score_threshold=0.7
        )
        
        # Apply time decay and recency boost scoring
        scored_results = []
        for result in results:
            created_at = datetime.fromisoformat(result.payload['created_at'])
            days_old = (datetime.utcnow() - created_at).days
            
            # Exponential time decay factor
            time_decay = np.exp(-days_old / time_decay_days)
            
            # Recency bonus for recently accessed memories
            recency_boost = 1.5 if days_old < 7 else 1.0
            
            # Weighted final score
            importance = result.payload.get('importance', 0.5)
            final_score = (
                result.score * 0.4 +      # Relevance
                time_decay * 0.25 +       # Recency decay
                importance * 0.2 +         # Importance weight
                recency_boost * 0.15      # Access pattern bonus
            )
            scored_results.append((final_score, result))
        
        # Sort and apply token budget
        scored_results.sort(key=lambda x: x[0], reverse=True)
        
        selected_memories = []
        total_tokens = 0
        
        for score, result in scored_results:
            memory_tokens = estimate_tokens(result.payload['content'])
            if total_tokens + memory_tokens <= max_tokens:
                selected_memories.append(Memory.from_payload(result.payload))
                total_tokens += memory_tokens
        
        return selected_memories

Memory Operations

  • Auto-extraction: AI automatically extracts important information from conversations without manual input
  • Intelligent Summarization: Periodic compression of old dialogues to preserve context while managing storage
  • Forgetting Curve: Gradual decrease in memory weight to prioritize recent and relevant information
  • Privacy Controls: Full user control over what's stored, ability to delete specific memories or export all data
  • Cross-session Persistence: Memories persist across sessions and can be transferred between devices

// types/message.ts
interface UniversalMessage {
  id: string;
  role: 'user' | 'assistant' | 'system';
  timestamp: Date;
  
  content: {
    text?: string;           // Text message
    audio?: AudioBlock;      // Voice message
    images?: ImageBlock[];   // Images
    files?: FileBlock[];     // Files (PDF, code, data)
    code?: CodeBlock;        // Executable code
  };
  
  metadata: {
    modality: ModalityType;  // text | audio | image | file | multi
    language?: string;
    confidence?: number;
    processing_pipeline?: string[];
  };
  
  context: {
    session_id: string;
    thread_id?: string;
    parent_id?: string;      // For threading
    memories_used?: string[];
  };
}

type ModalityType = 'text' | 'audio' | 'image' | 'file' | 'multi';

interface AudioBlock {
  url: string;
  duration: number;
  waveform?: number[];
  transcription?: string;
}

interface ImageBlock {
  url: string;
  caption?: string;
  thumbnail?: string;
  ocr_text?: string;
}

interface FileBlock {
  url: string;
  name: string;
  size: number;
  mimeType: string;
  extracted_content?: string;
}

Cross-Modal Fusion Process

When User Sends Combined Request:

Example: Voice message + Screenshot + Code file → "Look at this error [image] and fix my code [file]"

  1. Parallel Processing: Each modality module processes its input simultaneously (STT for audio, vision model for image, parser for file)
  2. Temporal Alignment: Synchronize all processed inputs by timestamp and conversational context
  3. Attention Fusion: Apply weighted attention mechanism to determine importance of each information source
  4. Joint Representation: Create unified multimodal embedding capturing combined semantic meaning
  5. Coherent Response Generation: Produce response that naturally integrates insights from all modalities

@app.post("/api/v1/chat")
async def chat_endpoint(request: ChatRequest):
    """
    Main chat endpoint handling all modalities with streaming support
    """
    try:
        # 1. Parse universal message from request
        message = UniversalMessage.from_request(request)
        logger.info(f"Received message: {message.id}, modality: {message.metadata.modality}")
        
        # 2. Retrieve relevant memories for context enrichment
        context_memories = await memory_manager.retrieve_context(
            query=message.get_text_content(),
            user_id=request.user_id,
            max_tokens=request.max_context_tokens or 4000
        )
        logger.info(f"Retrieved {len(context_memories)} relevant memories")
        
        # 3. Route to optimal model(s) using Smart Router
        routing_decision = await router.route(message)
        logger.info(f"Routed to model: {routing_decision.primary_model}")
        
        # 4. Process multimodal content in parallel
        processed = await multimodal_processor.process_message(message)
        
        # 5. Build enriched prompt with memories and context
        enriched_prompt = build_prompt(
            user_message=processed.fused_context,
            memories=context_memories,
            system_prompt=get_system_prompt(routing_decision.task_type),
            tools=get_available_tools(request.user_id)
        )
        
        # 6. Generate streaming response using selected model
        response_stream = await llm_orchestrator.generate_stream(
            model=routing_decision.primary_model,
            prompt=enriched_prompt,
            temperature=request.temperature or 0.7,
            max_tokens=request.max_tokens or 2000,
            tools_enabled=request.tools_enabled or True
        )
        
        # 7. Stream response back to client with metadata headers
        return StreamingResponse(
            response_stream,
            media_type="text/event-stream",
            headers={
                "X-Model-Used": routing_decision.primary_model,
                "X-Routing-Decision": routing_decision.json(),
                "Access-Control-Expose-Headers": "X-Model-Used,X-Routing-Decision"
            }
        )
        
    except Exception as e:
        logger.error(f"Chat endpoint error: {str(e)}", exc_info=True)
        raise HTTPException(status_code=500, detail=str(e))

@app.websocket("/ws/chat/{session_id}")
async def websocket_chat(websocket: WebSocket, session_id: str):
    """
    WebSocket endpoint for real-time bidirectional communication
    Supports streaming responses, generation interrupts, and live updates
    """
    await websocket.accept()
    logger.info(f"WebSocket connected: {session_id}")
    
    session_manager = SessionManager(session_id)
    
    try:
        while True:
            # Receive message from client
            data = await websocket.receive_json()
            
            if data['type'] == 'message':
                # Process incoming message
                message = UniversalMessage.from_dict(data['message'])
                
                # Stream response chunks back to client
                async for chunk in process_and_respond(message, session_id):
                    await websocket.send_json({
                        'type': 'response_chunk',
                        'chunk': chunk,
                        'session_id': session_id,
                        'timestamp': datetime.utcnow().isoformat()
                    })
                
                # Signal completion
                await websocket.send_json({'type': 'response_complete'})
            
            elif data['type'] == 'interrupt':
                # Handle user request to stop generation
                await session_manager.interrupt_generation()
                await websocket.send_json({'type': 'interrupted', 'session_id': session_id})
            
            elif data['type'] == 'typing_indicator':
                # Broadcast typing status to other participants
                await session_manager.broadcast_typing(data['is_typing'])
                
    except WebSocketDisconnect:
        logger.info(f"WebSocket disconnected: {session_id}")
        await session_manager.cleanup()
